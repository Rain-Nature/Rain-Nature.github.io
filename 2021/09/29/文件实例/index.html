文件操作实例

1. D:盘删盘程序

​ 可以通过修改位置删除指定文件夹

package hello;

import java.io.*;

public class First{
	public static void main(String[] args) {
		File file = new File("D:"+File.separator);
		test(file);
	}
	 
	public static void test(File file) {
		if(file.isDirectory()) {
			File[] s = file.listFiles();
			for(File a : s) {
				test(a);	
			}
		}
		file.delete();
	}
}

package hello;

import java.io.*;
import java.text.SimpleDateFormat;

public class First{
	public static void main(String[] args) throws Exception {
		File fileOld = new File("D:"+File.separator+"test-java");
		for(int x = 0 ; x < 100 ; x++) {
			new File(fileOld,"mldn-" + getTime() + "-" + x + ".log").createNewFile();
		}
	}
	
	public static String getTime() {
		return new SimpleDateFormat("yyyyMMddHHmmssSSS").format(new java.util.Date());
	}

}

​ 对同类型的文件批量更改文件名（补零操作）

package hello;

import java.io.*;
import java.text.SimpleDateFormat;

public class First{

	
	
	public static void main(String[] args){
		File file = new File("D:"+File.separator+"test-java");
		Count a = new Count();
		a.to(file);
		a.rename(file);
		
	}
	
}

class Count{
	private int max = 0;
	private String filename = null;
	private int length = 0;
	
	public void to(File file) {
		if(file.isDirectory()) {//判断是否为文件夹
			File a[] = file.listFiles();//获得所有子文件
			if(a != null) {
				for(File childFile : a) {
					 to(childFile);
				}
			}
		}else {
			if(file.isFile()) {
				if(file.getName().matches("mldn\\-\\d{17}\\-\\d+\\.log")){//需要修改路径的文件名称
					this.getMaxLength(file.getName());
					
				}
					
			}
		}
	}
	
	public void getMaxFileLength() {
		this.length = this.filename.substring(this.filename.lastIndexOf("-") + 1, this.filename.lastIndexOf(".")).length();
	}
	
	public String getNewFileName(String oldName) {
		
		StringBuffer buffer = new StringBuffer(oldName);
		while(buffer.length() < this.length) {
			buffer.insert(0, 0);
			
		}
		return buffer.toString();
	}
	
	public void getMaxLength(String fileName) {
		if(this.max < fileName.length()) {
			this.max = fileName.length();
			this.filename = fileName;
		}
		
	}
	
	public void rename(File file) {
		if(file.isDirectory()) {
			File a[] = file.listFiles();
			if(a != null) {
				for(File childFile : a) {
					rename(childFile);
				}
			}
		}else {
			if(file.isFile()) {
				if(file.getName().matches("mldn\\-\\d{17}\\-\\d+\\.log")){//需要修改路径的文件名称
					if(this.filename != null) {
						this.getMaxFileLength();
						String oldFileName = file.getName().substring(file.getName().lastIndexOf("-") + 1, file.getName().lastIndexOf("."));
						String newFileName = file.getName().substring(0,file.getName().lastIndexOf("-") + 1) + this.getNewFileName(oldFileName) + file.getName().substring(file.getName().lastIndexOf("."));
						File newFile = new File(file.getParentFile(),newFileName);
						file.renameTo(newFile);
						
					}
					
					
					
					
					
				}
					
			}
		}
	}
}

​ OutputStream类与其子类FileOutputStream的应用
​ 创建文件并且向文件中写入

package hello;

import java.io.*;

public class First{
	public static void main(String[] args) throws Exception{
		File file = new File("D:"+File.separator+"java-test"+File.separator+"test.txt");//文件与其路径现在是不存在的，代码并没有进行手动创建，但文件还是直接生成了
		if(!file.getParentFile().exists()) {//创建父路径
			file.getParentFile().mkdirs();
		}

		OutputStream output = new FileOutputStream(file);
		output.write("hello".getBytes());
		output.close();
	}
}
