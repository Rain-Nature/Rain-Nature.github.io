数据结构03

数据结构
Publish Date:   2022-10-31

链表

1. typedef

​ C语言允许为一个数据类型起一个新的别名，使用typedef关键字来进行定义

​ typedef 的用法一般为：

typedef  oldName  newName;

​ 注意：一般新名字全部字母大写

​ 范例：为结构体起一个别名

typedef struct Student{
	int sid;
	char name[100];
	char sex;
}ST,*PST;//ST等价于struct Student, PST等价于struct Student *

int main(void){
    ST st;
    st.sid = 200;
    printf("%d\n",st.sid);
}

2. 链表(离散存储)

链表的定义：

​ ​ 1. n个结点离散分配，
​ ​ 2. 彼此通过指针相连
​ ​ 3. 每个结点只有一个前驱结点，每个结点只有一个后续结点
​ ​ 4. 首结点没有前驱结点，尾结点没有后续结点

专业术语：

  • 首结点：第一个存了有效数据的结点
  • 尾结点：最后一个存了有效数据的结点，尾结点的指针域为空
  • 头结点：第一个有效结点(首结点)之前的结点，方便我们对链表算法进行操作，里面不包含有效数据和数组长度，但头结点的数据类型和首结点的数据类型一样
  • 头指针：指向头结点的指针变量，存放了头结点的地址
  • 尾指针：指向尾结点的指针变量

如果希望通过一个函数来对链表进行处理，至少需要接受链表的哪些参数：

​ 只需要一个参数：头指针

​ 通过头指针可以推算出链表的其他所有信息

3. 链表结点数据类型的表示

​ 一个结点可以分为两个部分：数据域和指针域

​ 指针域中的指针指向了一个完整的结点

​ 范例：结点的定义

typedef struct Node{
	int data;  //数据域
	struct Node * pNext; //指针域
}NODE,*PNODE;

4. 链表的分类

  • 单链表：每个结点的指针域只能指向后面结点
  • 双链表：每一个结点有两个指针域
  • 循环链表：能通过任何一个结点找到其他所有的结点
  • 非循环链表：尾结点为null

​

5. 链表的伪算法

  1. 非循环单链表：将q结点插入到p结点后面

​ 伪算法

//第一种方式
r = p->pNext;
p->pNext = q;
q->pNext = r;

//第二种方式
q->pNext = p->pNext;
p->pNext = q;

  1. 非循环单链表：将p结点后面的那个结点删除

​ 伪算法

//注意:由于直接连会导致中间那个结点找不到从而导致内存的浪费,所以要先存起来,最后也要进行释放
r = p->pNext;
p->pNext = p->pNext->pNext;
free(r);

​

6. 链表的算法

  1. 算法–链表的创建以及链表的遍历

    #include <stdio.h>
#include <stdlib.h>
#include <malloc.h>

//定义链表
typedef struct count{
    int number;
    struct count * pNext;
}NODE,* PNODE;

//函数声明
PNODE count_create(void);
void count_write(PNODE pHead);

int main()
{
    //设置头指针
    PNODE pHead = NULL;
    pHead = count_create(); //创建一个非循环单链表，并返回头结点地址
    count_write(pHead); //遍历打印出来
    return 0;
}

PNODE count_create(void){
    int i=0;
    int tmp; //临时存放用户输入的数据
    int num; //存放有效结点的个数

    //设置头指针指向头结点，头结点不存放任何有效数据
    PNODE pHead = (PNODE)malloc(sizeof(NODE));
    //头结点的指针域设为NULL，防止后面没有结点
    pHead->pNext = NULL;
    if(NULL == pHead){
        printf("error");
        exit(-1);
    }
    //链表的尾指针，用来挂每一个新生成的链表结点
    PNODE pTail = pHead;

    printf("please input number :");
    scanf("%d",&num);

    for(i=0; i < num ; i++){
        printf("The %d number:",i+1);
        scanf("%d",&tmp);
        PNODE new_Node = (PNODE) malloc(sizeof(NODE));
        if(NULL == new_Node){
            printf("error");
            exit(-1);
        }
        //新生成结点的指针域设置为NULL
        new_Node->pNext = NULL;
        //新生成的结点的数据域设置值
        new_Node->number =tmp;
        //将新生成的结点挂在尾指针上
        pTail->pNext = new_Node;
        //将尾指针指向新生成的新结点
        pTail = new_Node;
    }
    return pHead;
}

void count_write(PNODE pHead){
    PNODE pTail = pHead->pNext;
    while(NULL!=pTail){
        printf("%d" ,pTail->number);
        //将指针向后移
        pTail = pTail->pNext;
    }
}

  1. 算法–链表的是否为空的判断和求链表长度

    //判断是否为空的函数
#include <stdbool.h>  //C语言使用bool类型需要导入的头文件
bool is_empty(PNODE pHead); //函数声明
bool is_empty(PNODE pHead){ //函数体
    if(NULL == pHead->pNext){
        return true;
    }
    return false;
}

//求链表长度
int count_len(PNODE pHead); //函数声明
int count_len(PNODE pHead){ //函数体
    int len = 0;
    PNODE p = pHead->pNext;
    while(NULL != p){
        ++len;
        p = p->pNext;
    }
    return len;
}

  1. 算法–链表的冒泡排序

    //冒泡排序
void count_sort(PNODE pHead); //函数声明
void count_sort(PNODE pHead){ //函数体
    int i,j,tmp;
    int len = count_len(pHead);
    PNODE p = pHead->pNext;

    for(i = 0; i<len-1 ; i++){
        PNODE q = p->pNext;
        for(j = i+1; j < len ; j++){
            if(p->number>q->number){
                tmp = p->number;
                p->number = q->number;
                q->number = tmp;

            }
            q = q->pNext;
        }
        p = p->pNext;
    }
}

  1. 算法–链表的插入元素和删除元素

    //链表的插入算法
bool count_insert(PNODE pHead, int pos , int val); //函数声明 pos表示在有效结点第pos个增加结点
bool count_insert(PNODE pHead, int pos , int val){ //函数体
    int i;
    if(pos<0){
        return false;
    }
    PNODE p = pHead;
    for(i = 0; i<pos; i++){
        p = p->pNext;
        //一定要判空，否则程序有bug
        if(p==NULL){
            return false;
        }
    }
    PNODE pNew = (PNODE)malloc(sizeof(NODE));
    pNew->number = val;
    pNew->pNext = p->pNext;
    p->pNext = pNew;
    return true;
}

//链表的删除算法
bool count_delete(PNODE pHead, int pos , int *val); //函数声明，pos表示删除有效结点的第pos个
bool count_delete(PNODE pHead, int pos , int *val){
    int i;
    if(pos<1){
        return false;
    }
    PNODE p = pHead;
    for(i = 1 ; i<pos ;i++){
        if(p->pNext==NULL){
            return false;
        }
        p = p->pNext;
    }
    if(p->pNext==NULL){
        return false;
    }
    PNODE tmp = p->pNext;
    *val = tmp->number;
    //将p指针指向下下个结点
    p->pNext = p->pNext->pNext;
    free(tmp);
    return true;

}
